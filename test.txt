repeat task.wait() until game:IsLoaded()

local player = game:GetService("Players").LocalPlayer
local HttpService = game:GetService("HttpService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local CommF = ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("CommF_")

getgenv().BlockXero = false

local CONFIG = getgenv().CONFIG or {
    ["CONFIG"] = {
        ["Find Tushita"] = true,
        ["Find Valkyrie Helm"] = true,
        ["Find Dark Fragment"] = true,
        ["Find Cursed Captain"] = true
    }
}

local loadedScript = false

local function getSea()
    if workspace:FindFirstChild("Map") then
        if workspace.Map:FindFirstChild("Dressrosa") then
            return 2
        elseif workspace.Map:FindFirstChild("TikiOutpost") then
            return 3
        else
            return 1
        end
    end
    return 0
end

local function hasItem(name)
    local success, result = pcall(function()
        local inv = CommF:InvokeServer("getInventory")
        for _, v in pairs(inv) do
            if v.Name == name then
                return v.Count or 1
            end
        end
        return 0
    end)
    return success and result or 0
end

local function getPlayerLevel()
    local success, lvl = pcall(function()
        return player.Data.Level.Value
    end)
    return success and lvl or 0
end

local function hasBoss(name)
    local enemies = workspace:FindFirstChild("Enemies")
    if not enemies then return false end

    for _, v in pairs(enemies:GetChildren()) do
        if v:FindFirstChild("Humanoid") then
            if v.Humanoid.MaxHealth > 200000 then
                if string.lower(v.Name):find(string.lower(name)) then
                    return true
                end
            end
        end
    end

    return false
end

local function GetCDKProcess()
    local CDKPuzzle = {Evil = 0, Good = 0}
    local data = CommF:InvokeServer("CDKQuest", "Progress", "Good")
    for i, v in next, data do
        CDKPuzzle[i] = v
    end

    local Evil = CDKPuzzle.Evil
    local Good = CDKPuzzle.Good

    if Good ~= -2 and Good ~= 3 and Good ~= 4 then
        if Good == 2 or Good == -5 then
            return "Heavenly Dimension Quest"
        elseif Good == 1 or Good == -4 then
            return "Raid Castle Quest"
        elseif Good == 0 or Good == -3 then
            return "Boat Quest"
        end
    else
        if Evil == 2 or Evil == -5 then
            return "Hell Dimension Quest"
        elseif Evil == 1 or Evil == -4 then
            return "Haze Quest"
        elseif Evil == 0 or Evil == -3 then
            return "Die Quest"
        end
    end

    return "None"
end

local function setupConfig(mode)
    local folder = "Night Hub Hop Rewrite"
    local file = folder .. "/" .. player.Name .. ".json"

    local data = {
        ["Select Tool"] = "Melee",
        ["Auto Execute Script"] = false,
        ["Auto Tushita"] = false,
        ["Auto Tushita [HOP]"] = false,
        ["Auto Kill Boss"] = false,
        ["Auto Kill Boss [HOP]"] = false,
        ["Select Boss"] = "rip_indra"
    }

    if mode == "TUSHITA" then
        data["Auto Tushita"] = true
        data["Auto Tushita [HOP]"] = true
    elseif mode == "INDRA" then
        data["Auto Kill Boss"] = true
        data["Auto Kill Boss [HOP]"] = true
    elseif mode == "DARKBEARD" then
        data["Auto Kill Boss"] = true
        data["Auto Kill Boss [HOP]"] = true
        data["Select Boss"] = "Darkbeard"
    elseif mode == "CDK" then
        data["Auto Kill Boss"] = true
        data["Auto Kill Boss [HOP]"] = true
        data["Select Boss"] = "Cursed Captain"
    end

    if makefolder then makefolder(folder) end
    if writefile then
        writefile(file, HttpService:JSONEncode(data))
    end
end

local function runScript()
    if not loadedScript then
        loadedScript = true
        getgenv().Team = "Marines"
        loadstring(game:HttpGet("https://raw.githubusercontent.com/WhiteX1208/Scripts/refs/heads/main/HopScript.luau"))()
    end
end

-- 🔴 DARK FRAGMENT (CHỈ CHECK 1 LẦN)
local function checkDarkFragmentOnce()
    local cfg = CONFIG["CONFIG"]
    local level = getPlayerLevel()
    local sea = getSea()

    if cfg["Find Dark Fragment"] and level > 1500 and sea == 2 then
        if hasItem("Dark Fragment") == 0 then
            getgenv().BlockXero = true
            setupConfig("DARKBEARD")
            runScript()
        else
            getgenv().BlockXero = false
        end
    end
end

checkDarkFragmentOnce()

-- 🟡 CDK / Cursed Captain (30s)
task.spawn(function()
    while true do
        task.wait(30)

        local level = getPlayerLevel()
        if level <= 2000 then continue end

        local sea = getSea()
        local cfg = CONFIG["CONFIG"]

        if not cfg["Find Cursed Captain"] then continue end
        if sea ~= 3 then continue end

        local alucard = hasItem("Alucard Fragment")

        if alucard >= 3 and GetCDKProcess() == "Hell Dimension Quest" then
            setupConfig("CDK")

            if not hasBoss("Cursed Captain") then
                runScript()
            else
                task.delay(300, function()
                    runScript()
                end)
            end
        end
    end
end)

-- ⚡ TUSHITA + VALK (60s)
task.spawn(function()
    while true do
        task.wait(60)

        local level = getPlayerLevel()
        if level <= 2000 then continue end

        local sea = getSea()
        local cfg = CONFIG["CONFIG"]

        if sea ~= 3 then continue end

        local tushita = hasItem("Tushita") > 0
        local valk = hasItem("Valkyrie Helm") > 0

        if cfg["Find Tushita"] and not tushita then
            setupConfig("TUSHITA")
            runScript()
            continue
        end

        if cfg["Find Valkyrie Helm"] and not valk then
            setupConfig("INDRA")
            runScript()
        end
    end
end)
